#!/usr/bin/python3


# Test case 1: Empty list
>>> my_list = MyList()
>>> print(my_list)
[]
>>> my_list.print_sorted()
[]
>>> print(my_list)
[]

# Test case 2: List with one element
>>> my_list = MyList()
>>> my_list.append(1)
>>> print(my_list)
[1]
>>> my_list.print_sorted()
[1]
>>> print(my_list)
[1]

# Test case 3: List with multiple elements
>>> my_list = MyList()
>>> my_list.append(1)
>>> my_list.append(4)
>>> my_list.append(2)
>>> my_list.append(3)
>>> my_list.append(5)
>>> print(my_list)
[1, 4, 2, 3, 5]
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]
>>> print(my_list)
[1, 4, 2, 3, 5]

# Test case 4: List with duplicate elements
>>> my_list = MyList()
>>> my_list.append(1)
>>> my_list.append(1)
>>> my_list.append(2)
>>> my_list.append(2)
>>> my_list.append(3)
>>> my_list.append(3)
>>> print(my_list)
[1, 1, 2, 2, 3, 3]
>>> my_list.print_sorted()
[1, 1, 2, 2, 3, 3]
>>> print(my_list)
[1, 1, 2, 2, 3, 3]

# Test case 5: List with negative elements
>>> my_list = MyList()
>>> my_list.append(-1)
>>> my_list.append(-4)
>>> my_list.append(-2)
>>> my_list.append(-3)
>>> my_list.append(-5)
>>> print(my_list)
[-1, -4, -2, -3, -5]
>>> my_list.print_sorted()
[-5, -4, -3, -2, -1]
>>> print(my_list)
[-1, -4, -2, -3, -5]
